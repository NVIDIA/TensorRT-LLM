def is_deepseek_r1_1k1k_or_8k1k(case: DisaggCase, path: Path) -> bool:
    name_lower = case.model_name.lower()
    if "deepseek-r1" not in name_lower and "deepseek_r1" not in name_lower:

def aggregate_generation():
    qa_disagg = scan_qa_disagg()
    ctx_groups = {}
    for p, y in qa_disagg:
        key = canonical_ctx_key(y)
        ctx_groups.setdefault(key, []).append((p, y))

    generated = []
    dedup_groups = []
    skipped = []

    for key, members in ctx_groups.items():
        if len(members) > 1:
            dedup_groups.append({
                "ctx_key": key,
                "members": [str(m[0]) for m in members],
            })
        # Per instructions, one agg YAML per unique (ctx, isl/osl) pair = our key.
        existing = scan_qa_agg_for_ctx_key_match(key)
        if existing:
            skipped.append({
                "yaml": str(existing),
                "ctx_key": key,
                "reason": "ctx key already covered by existing agg case",
            })
            continue

        # Generate new agg YAML
        src_path, src_yaml = members[0]
        filename = derive_agg_filename(src_yaml)
        agg_path = QA_AGG_DIR / filename
        y_out = build_agg_yaml_from_ctx(src_yaml, agg_path)

        # Write YAML with minimal header comment
        hdr = textwrap.dedent(f"""\
        # Copyright (c) {datetime.datetime.now().year}, NVIDIA CORPORATION.
        # Licensed under the Apache License, Version 2.0 (the "License");
        """)
        yaml_text = yaml.safe_dump(y_out, sort_keys=False)
        agg_path.write_text(hdr + yaml_text)

        generated.append({
            "name": agg_path.name,
            "path": str(agg_path),
            "derived_from": [str(m[0]) for m in members],
            "ctx_key": key,
        })

    return generated, dedup_groups, skipped

def parse_test_list_for_cases():
    lines = read_test_list()
    cases = []
    current_section = None
    for line in lines:
        s = line.strip()
        if s.startswith("#"):
            current_section = s
            continue
        if s.startswith("perf/test_perf_sanity.py::test_"):
            m = re.search(r"\[(.+?)\]", s)
            if not m:
                continue
            test_id = m.group(1)
            mode = "e2e"
            if test_id.startswith("disagg-gen_only"):
                mode = "gen_only"
            elif test_id.startswith("aggr-ctx_only"):
                mode = "agg"
            cases.append({
                "section": current_section,
                "test_id": test_id,
                "mode": mode,
            })
    return cases

def resolve_yaml_for_test_id(test_id: str):
    # disagg: disagg-...-<cfg>
    # aggr:   aggr-ctx_only-<cfg>
    if test_id.startswith("aggr-ctx_only-"):
        cfg = test_id[len("aggr-ctx_only-"):]
        p = QA_AGG_DIR / f"{cfg}.yaml"
        return p, "agg"
    # assume all others are disagg
    # find last '-' that separates prefix and cfg
    m = re.match(r"(disagg-[^[]*?)-(.*)", test_id)
    if m:
        cfg = m.group(2)
    else:
        # fallback: test_id itself is cfg
        cfg = test_id
    p = QA_DISAGG_DIR / f"{cfg}.yaml"
    return p, "disagg"

def compute_gpu_hours_for_case(yaml_path: Path, mode: str):
    y = load_yaml(yaml_path)
    if mode == "agg":
        sc = (y.get("server_configs") or [])[0]
        tp = int(sc.get("tensor_parallel_size", 1))
        pp = int(sc.get("pipeline_parallel_size", 1))
        cp = int(sc.get("context_parallel_size", 1))
        total_gpus = tp * pp * cp
    else:
        hw = y.get("hardware", {})
        wc = y.get("worker_config", {})
        ctx = wc.get("ctx", {})
        gen = wc.get("gen", {})
        nctx = int(hw.get("num_ctx_servers", 1))
        ngen = int(hw.get("num_gen_servers", 1))
        ctx_tp = int(ctx.get("tensor_parallel_size", 1))
        ctx_pp = int(ctx.get("pipeline_parallel_size", 1))
        ctx_cp = int(ctx.get("context_parallel_size", 1))
        gen_tp = int(gen.get("tensor_parallel_size", 1))
        gen_pp = int(gen.get("pipeline_parallel_size", 1))
        gen_cp = int(gen.get("context_parallel_size", 1))
        if mode == "gen_only":
            total_gpus = ngen * (gen_tp * gen_pp * gen_cp)
        else:
            total_gpus = nctx * (ctx_tp * ctx_pp * ctx_cp) + ngen * (gen_tp * gen_pp * gen_cp)

    gpu_hours = total_gpus * 1.0  # 1 h duration assumption
    nodes = math.ceil(total_gpus / 4.0)
    return gpu_hours, nodes

For a disaggregated **e2e** case, GPU-hours are computed from the ctx and gen sides separately, then summed:

1. **Extract from YAML** (disagg schema):
   - From `hardware`:
     - `num_ctx_servers`
     - `num_gen_servers`

     - `pipeline_parallel_size` (PP)
     - `context_parallel_size` (CP, defaults to 1 if missing)

     - `pipeline_parallel_size` (PP)

   - Context GPUs per server:
     `ctx_gpus_per_server = ctx.TP × ctx.PP × ctx.CP`
   - Generation GPUs per server:
     `gen_gpus_per_server = gen.TP × gen.PP × gen.CP`

3. **Compute total GPUs used by the test**:
   - Context side total:
     `ctx_total_gpus = num_ctx_servers × ctx_gpus_per_server`
   - Gen side total:
     `gen_total_gpus = num_gen_servers × gen_gpus_per_server`
   - Combined:
     `total_gpus = ctx_total_gpus + gen_total_gpus`

4. **Convert to GPU-hours** (assuming 1 hour runtime per case):
   - **disagg e2e GPU-hours**:
     ```text
     GPU_hours = num_ctx_servers × (ctx.TP × ctx.PP × ctx.CP)
               + num_gen_servers × (gen.TP × gen.PP × gen.CP)
     ```

   - Context: `1 × (4×1×1) = 4`
